**Scenario A — Target workspace active in another window (cross-window move):**
1. Set `isSwitching: true` via `setSessionState` (prevents throttledSave from interfering — Pitfall 2 from research)
2. Snapshot workspaces for rollback: `const snapWorkspaces = JSON.parse(JSON.stringify(workspaces))`
3. Sort tab IDs so pinned tabs come first to prevent silent move failures (Pitfall 3): `const sortedTabs = [...tabs].sort((a, b) => (b.pinned ? 1 : 0) - (a.pinned ? 1 : 0))`
4. Call `await browser.tabs.move(sortedTabs.map(t => t.id), { windowId: Number(targetWindowId), index: -1 })` — physically moves tabs without reload (D-05)
5. Update source workspace tabs: filter out moved tabs by tab ID set
6. Re-query target window tabs and serialize: `const targetWindowTabs = await browser.tabs.query({ windowId: Number(targetWindowId) })` then `targetWs.tabs = serializeTabs(targetWindowTabs)`
7. Set `targetWs.lastUsedAt = Date.now()`
8. Save workspaces, then update badge for source window
9. Focus target window: `await browser.windows.update(Number(targetWindowId), { focused: true })` (D-02/D-03)

**Scenario B — Target workspace inactive (same-window move + switch):**
1. Set `isSwitching: true`
2. Snapshot for rollback
3. Serialize moved tabs: `const movedTabData = serializeTabs(tabs)`
4. Append to target: `targetWs.tabs = [...targetWs.tabs, ...movedTabData]`
5. Remove from source: filter out moved tab URLs from `sourceWs.tabs`
6. Set `targetWs.lastUsedAt = Date.now()`
7. Save workspaces
8. Call `await switchWorkspace(targetWsId, sourceWindowId)` to display target workspace (D-06/D-07 — using reload approach per D-07 permission)

**Shared structure:**

```javascript
export async function moveTabsToWorkspace(tabs, targetWsId, sourceWindowId) {
  await setSessionState({ isSwitching: true })
  let snapWorkspaces = null

  try {
    const workspaces = await getWorkspaces()
    const windowMap = await getWindowMap()

    const targetWs = workspaces.find(w => w.id === targetWsId)
    if (!targetWs) return { success: false, error: 'Target workspace not found' }

    const sourceWsId = windowMap[String(sourceWindowId)]
    const sourceWs = workspaces.find(w => w.id === sourceWsId)

    // Snapshot for rollback (D-09)
    snapWorkspaces = JSON.parse(JSON.stringify(workspaces))

    // Find if target workspace is active in another window
    const targetWindowEntry = Object.entries(windowMap)
      .find(([wid, wsId]) => wsId === targetWsId && wid !== String(sourceWindowId))
    const targetWindowId = targetWindowEntry ? Number(targetWindowEntry[0]) : null

    if (targetWindowId) {
      // Cross-window: physically move tabs (D-05, no reload)
      // Sort pinned tabs first to avoid silent move failures (Pitfall 3)
      const sortedTabs = [...tabs].sort((a, b) => (b.pinned ? 1 : 0) - (a.pinned ? 1 : 0))
      await browser.tabs.move(sortedTabs.map(t => t.id), { windowId: targetWindowId, index: -1 })

      // Update source workspace tab list
      if (sourceWs) {
        const movedIds = new Set(tabs.map(t => t.id))
        const remainingTabs = await browser.tabs.query({ windowId: sourceWindowId })
        sourceWs.tabs = serializeTabs(remainingTabs)
      }

      // Update target workspace tab list from live window state
      const targetWindowTabs = await browser.tabs.query({ windowId: targetWindowId })
      targetWs.tabs = serializeTabs(targetWindowTabs)
      targetWs.lastUsedAt = Date.now()

      await saveWorkspaces(workspaces)

      // Update badge for source window
      if (sourceWs) updateBadge(sourceWs, sourceWindowId)

      // Focus target window (D-02/D-03)
      await browser.windows.update(targetWindowId, { focused: true })
    } else {
      // Same-window: save moved tabs into target, remove from source, then switch
      const movedTabData = serializeTabs(tabs)
      targetWs.tabs = [...targetWs.tabs, ...movedTabData]

      if (sourceWs) {
        const movedUrls = new Set(tabs.map(t => t.url))
        sourceWs.tabs = sourceWs.tabs.filter(t => !movedUrls.has(t.url))
      }

      targetWs.lastUsedAt = Date.now()
      await saveWorkspaces(workspaces)
      await switchWorkspace(targetWsId, sourceWindowId)
    }

    return { success: true }
  } catch (e) {
    console.error('[Workspaces] Move error:', e)
    if (snapWorkspaces) {
      await saveWorkspaces(snapWorkspaces).catch(err =>
        console.error('[Workspaces] Move rollback failed:', err)
      )
    }
    return { success: false, error: e.message }
  } finally {
    await setSessionState({ isSwitching: false })
  }
}
```

**Important details:**
- The `_tabId` unused parameter convention from eslint config applies if any callback params are unused
- Use `const movedIds = new Set(tabs.map(t => t.id))` for cross-window removal (by ID, since tabs are still live)
- Use `const movedUrls = new Set(tabs.map(t => t.url))` for same-window removal (tabs may not have stable IDs after switch)
- For cross-window path, re-query source window tabs after move to get accurate remaining list (safer than URL-based filtering, since tabs may have duplicate URLs)
- The `finally` block always resets `isSwitching` even if rollback fails
- Do NOT add a new import — `getWindowMap` and `setSessionState` are already imported in workspaces.js (line 3)
